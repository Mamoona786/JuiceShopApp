# 1. Create a new Next.js app (JavaScript version)
npx create-next-app@latest juice-shop-web --js --use-npm
cd juice-shop-web

# 2. Install React Router alternative (Next.js has built-in routing, so no need to install react-router-dom)
npm install react-icons

# 4. Install Axios
npm install axios

# 5. Install MDB React UI Kit (force to resolve peer deps if needed)
npm install mdb-react-ui-kit --force

# 6. Start the development server
npm run dev
